\section{WHIR-based VM security level calculation}\label{sec:whir_security_level}




\subsection{Notation and parameters}\label{sec:whir_notation}

\begin{itemize}
  \item Number of iterations: \(M = \texttt{num\_iterations}\)
  \item Iteration index: \(i \in \{0,\dots,M-1\}\)
  \item Folding factor in iteration \(i\): \(k_i = \texttt{folding\_factors}[i]\)
  \item Folding round index in iteration \(i\): \(s \in \{0,\dots,k_i-1\}\)
  \item Extension field size: \(|\mathbb{F}| = \bfield\)
  \item Constraint degree: \(d = \texttt{constraint\_degree}\)
  \item Log-degree in iteration \(i\): \(m_i = \texttt{log\_degrees}[i]\), and we set $m_{i+1} = m_i - k_i$ (equivalently $m_i = m_0 - \sum_{j=0}^{i-1} k_j$)
  \item Log-inverted rate in iteration \(i\): \(\mu_i = \texttt{log\_inv\_rates}[i]\), and we set $\mu_{i+1} = \mu_i + (k_i - 1)$ (equivalently $\mu_i = \log_2 \frac{1}{\rho} + \sum_{j=0}^{i-1}(k_j - 1)$)
  \item Iteration-round-specific RS codes:
  $$
  \CC_{i,s}=(\CC_{i,s}[\rho],\CC_{i,s}[N])_{i,s}=(2^{-\mu_i},2^{m_i-s})
  $$
  \item OOD sample counts (length \(M-1\)): \(w_j = \texttt{num\_ood\_samples}[j]\) for \(j\in\{0,\ldots,M-2\}\); iteration \(i\in\{1,\ldots,M-1\}\) sends \(w_{i-1}\) OOD evaluations.
  \item Number of queries in iteration \(i\): \(t_i = \texttt{num\_queries}[i]\) for \(i\in[M]\).
  \item Grinding bits:
    \[
      g_{\mathrm{batch}},\quad
      \{g^{\mathrm{fold}}_{i,s}\}_{i\in [M],\,s\in[k_i]},\quad
      \{g^{\mathrm{ood}}_j\}_{j\in\{0,\ldots,M-2\}},\quad
      \{g^{\mathrm{qry}}_i\}_{i\in[M]}.
    \]
\end{itemize}



\subsection{Bits of Security in UDR and JBR}

In order to compute the bits of security in UDR and JBR regimes, we compute the following error terms using the parameters from \cref{sec:whir_notation}.

For any\footnote{See \texttt{pcs/whir.py:get\_pcs\_security\_levels()}.} error term:
\[
  \lambda(\epsilon)
  = \lfloor-\log_2\epsilon\rfloor.
\]

Overall security:
\[
  \lambda_{\mathrm{total}}
    = \min\{
      \lambda_{\mathrm{batch}},
      \{\lambda^{\mathrm{fold}}_{i,s}\}_{i\in [M],\,s\in[k_i]},
      \{\lambda^{\mathrm{out}}_i\}_{i\in \{1,\ldots,M-1\}},
      \{\lambda^{\mathrm{shift}}_i\}_{i\in \{1,\ldots,M-1\}},
      \lambda^{\mathrm{fin}}
    \}.
\]

\subsection{Batching Error \(\epsilon_{\mathrm{batch}}\)}

The batching error\footnote{See \texttt{pcs/whir.py:\_get\_batching\_error()}.} uses parameters from \cref{sec:whir_notation}.

In the Johnson Bound Regime (JBR) we proceed as follows:
\begin{itemize}
    \item Set code parameters $$
    (\rho,N)= \left(2^{-\mu_0}, 2^{m_0}\right)
    $$
    \item Compute base error (cf. \cref{sec:batch}):
\[
\varepsilon_{\mathrm{batch}, JBR}^{\text{base}} =
\begin{cases}
   \ebatchpowJ(\rho,N,\bfield,B) & \text{(power batching)}, \\[4pt]
  \ebatchlinJ(\rho,N,\bfield)& \text{(linear batching)}.
\end{cases}
\]
\end{itemize}
Whereas in the UDR we do as follows:
\begin{itemize}
    \item Compute base error (cf. \cref{sec:batch}):
\[
\varepsilon_{\mathrm{batch}, UDR}^{\text{base}} =
\begin{cases}
   \ebatchpowU & \text{(power batching)}, \\[4pt]
  \ebatchlinU& \text{(linear batching)}.
\end{cases}
\]
\end{itemize}

In both regimes we do as follows after grinding:
\[
  \epsilon_{\mathrm{batch}}
  = \varepsilon_{\mathrm{batch}}^{\text{base}}
    \cdot 2^{-g_{\mathrm{batch}}}.
\]
\subsection{Folding Error}

The folding error per round refers to Theorem 5.2 of the WHIR paper.\footnote{See \texttt{pcs/whir.py:\_epsilon\_fold()}.}

For iteration \(i\in[M]\) and folding round \(s\in[k_i]\) in JBR we do\footnote{The function \texttt{pcs/whir.py:\_epsilon\_fold()}
counts folding rounds from \(1\) to \(k_i\), whereas here we count from \(0\) to \(k_i-1\).    }:
\begin{itemize}
    \item % \verb|get_list_size_for_iteration_and_round()|
 %   \begin{itemize}
   % \item
    Get list size as in \cref{sec:fri_list_sizes}
    $$
    \ell_{i,s} =\ell(\CC_{i,s}[\rho])
    $$
  %  \end{itemize}
    \item
Base error (two terms):
\[
  \varepsilon^{\mathrm{fold,base,J}}_{i,s}
    = d \cdot \frac{\ell_{i,s}}{\bfield}
      + \ebatchpowJ(\CC_{i,s+1}[\rho],\CC_{i,s+1}[N],\bfield,B=2).
\]
\end{itemize} And in UDR the base error is

\[
  \varepsilon^{\mathrm{fold,base,U}}_{i,s}
    =  \frac{d}{\bfield}
      + \ebatchpowU(\CC_{i,s+1}[\rho],\CC_{i,s+1}[N],\bfield,B=2).
\]

After grinding:
\[
  \epsilon^{\mathrm{fold}}_{i,s}
    = \varepsilon^{\mathrm{fold,base}}_{i,s}
      \cdot 2^{-g^{\mathrm{fold}}_{i,s}}.
\]

\subsection{OOD error }

The Out-of-Domain error\footnote{See \texttt{pcs/whir.py:\_epsilon\_out()}.} is computed as follows.

For iteration \(i\in\{1,2,\ldots,M-1\}\)
\begin{itemize}
    \item
Base error in JBR (cf. \cref{sec:fri_list_sizes}):
\[
  \varepsilon^{\mathrm{out,base,J}}_i
  = \ell_{i,0}^2
    \left(\frac{2^{m_i}}{2q}\right)^{w_{i-1}}.
\]
\item Base error in UDR:
\[
  \varepsilon^{\mathrm{out,base,U}}_i
  =  \left(\frac{2^{m_i}}{2q}\right)^{w_{i-1}}.
\]
\item After grinding:
\[
  \epsilon^{\mathrm{out}}_i
  = \varepsilon^{\mathrm{out,base}}_i
    \cdot 2^{-g^{\mathrm{ood}}_{i-1}}.
\]

\end{itemize}
\subsection{Shift Error}

For iteration\footnote{See \texttt{pcs/whir.py:\_epsilon\_shift()}.} \(i\in\{1,2,\ldots,M-1\}\) in JBR
\begin{itemize}
    \item   Get $\delta$ for the iteration using \eqref{eq:pp_jbr}:
    $$
    \delta_i =\min_{s\in[k_{i-1}+1]}\delta_J(\CC_{i-1,s}[\rho],\CC_{i-1,s}[N],q)
    $$
\item Base error (two terms):
\[
  \varepsilon^{\mathrm{shift,base}}_i
    = (1 - \delta_i )^{t_{i-1}}
      + \ell_{i,0} \cdot \frac{t_{i-1} + 1}{  q}.
\]

\end{itemize}

Same procedure in UDR:
\begin{itemize}
    \item   Get $\delta$ for the iteration using \eqref{eq:pp_udr}:
    $$
    \delta_i =\min_{s\in[k_{i-1}+1]}\delta_U(\CC_{i-1,s}[\rho])
    $$
\item Base error  :
\[
  \varepsilon^{\mathrm{shift,base}}_i
    = (1 - \delta_i )^{t_{i-1}}
      +   \frac{t_{i-1} + 1}{q}.
\]

\end{itemize}

 After grinding:
\[
  \epsilon^{\mathrm{shift}}_i
    = \varepsilon^{\mathrm{shift,base}}_i
      \cdot 2^{-g^{\mathrm{qry}}_{i-1}}.
\]


\subsection{Final Round Error}

In JBR:\footnote{See \texttt{pcs/whir.py:\_epsilon\_final()}.}
\begin{itemize}
    \item  Get $\delta$ for the iteration using \eqref{eq:pp_jbr}:
    $$
    \delta_{M-1} =\min_{s\in[k_{M-1}+1]}\delta_J(\CC_{M-1,s}[\rho])
    $$
\end{itemize}
UDR:
 \begin{itemize}
    \item  Get $\delta$ for the iteration using \eqref{eq:pp_udr}:
    $$
    \delta_{M-1} =\min_{s\in[k_{M-1}+1]}\delta_U(\CC_{M-1,s}[\rho])
    $$
\end{itemize}
Then
the base error is :
\[
  \varepsilon^{\mathrm{fin,base}}
    = (1 - \delta_{M-1})^{t_{M-1}}.
\]
After grinding:
\[
  \epsilon^{\mathrm{fin}}
  = (1 - \delta_{M-1})^{t_{M-1}}
    \cdot 2^{-g^{\mathrm{qry}}_{M-1}}.
\]

\section{WHIR Proof Size Calculations}

This section summarizes the proof-size formula.\footnote{See \texttt{pcs/whir.py:get\_proof\_size\_bits()}.}

\subsection{Initial Function  Size}
\[
  S_{\mathrm{if}} = \bhash.
\]

\subsection{Sumcheck Size for Iteration \texorpdfstring{\(i\)}{i}}


The folding proof size at round $i$ is:
\[
  S_{\mathrm{sumcheck},i} = k_i(d-1)\bfield.
\]

\subsection{Merkle root and OOD Proof Size for Iteration \texorpdfstring{\(i\)}{i}}

For iterations \(i\in\{1,\ldots,M-1\}\), the prover sends a Merkle root for $f_i$ and the $w_{i-1}$ OOD evaluations:
\[
  S_{\mathrm{ood},i}
    = \bhash+w_{i-1}\bfield.
\]

After the last iteration the prover also sends the final multilinear polynomial:
$$
S_{\mathrm{final}}=2^{m_M}\bfield.
$$

\subsection{Query Proof Size for Iteration \texorpdfstring{\(i\)}{i}}


We compute
\[
  S_{\mathrm{qry},i}
    =\begin{cases}  eMP(2^{m_i+\mu_i-k_i},t_i,2^{k_i},\bfield,\bhash).\quad i>0\\
     eMP(2^{m_0+\mu_0-k_0},t_0, B\cdot 2^{k_0},\bprime,\bhash).\quad i=0
    \end{cases}
\]


\subsection{Total Proof Size}

Collecting all terms and summing over all \(M\) iterations:

\[
\boxed{
S_{\mathrm{total}}
  = S_{\mathrm{if}} + S_{\mathrm{final}}
    + \sum_{i\in[M]} \left( S_{\mathrm{sumcheck},i} + S_{\mathrm{qry},i} \right)
    + \sum_{i\in\{1,\ldots,M-1\}} S_{\mathrm{ood},i}
}
\]
